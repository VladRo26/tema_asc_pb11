#include <unistd.h>

#define len 30

char msg_da[] = "Este\n";
#define msg_da_len 5

char msg_nu[] ="Nu e\n";
#define msg_nu_len 5

char string[len];
char i;

int main()
{
    i = read(0, string, len);

    i -= 2;

    if(string[i] == '0' || string[i] == '5')
        write(1, msg_da, msg_da_len);
    else
        write(1, msg_nu, msg_nu_len);

    return 0;
}
